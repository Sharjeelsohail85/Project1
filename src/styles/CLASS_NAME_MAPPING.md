## Usage Example

```jsx
import styles from './Button.module.css';

function MyButton({ children }) {
  return (
    <button className={styles.button}>
      {children}
    </button>
  );
}

// For multiple classes:
import cn from 'classnames'; // optional helper

<button className={cn(styles.button, styles.buttonGoogle)}>
  Sign in with Google
</button>

// Or without classnames:
<button className={`${styles.button} ${styles.buttonGoogle}`}>
  Sign in with Google
</button>
```
