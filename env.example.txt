# API Configuration
# Copy this file to .env in the Project1 directory

# Laravel Backend API URL
# Update this to match your Laravel backend URL
# Default: http://localhost:8000/api/v1
VITE_API_BASE_URL=http://localhost:8000/api/v1

# OAuth Configuration (Optional - for OAuth login)
# Get credentials from respective providers

# Google OAuth
VITE_GOOGLE_CLIENT_ID=your_google_client_id_here
VITE_GOOGLE_REDIRECT_URI=http://localhost:3000/auth/google/callback

# Facebook OAuth
VITE_FACEBOOK_APP_ID=your_facebook_app_id_here
VITE_FACEBOOK_REDIRECT_URI=http://localhost:3000/auth/facebook/callback

# Dropbox OAuth
VITE_DROPBOX_CLIENT_ID=your_dropbox_client_id_here
VITE_DROPBOX_REDIRECT_URI=http://localhost:3000/auth/dropbox/callback
